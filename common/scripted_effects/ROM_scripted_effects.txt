ROM_remove_iron_guard_tension = {
	if = {
		limit = {
			has_idea = ROM_iron_guard_tension_4
		}
		remove_ideas = ROM_iron_guard_tension_4
	}
	else_if = {
		limit = {
			has_idea = ROM_iron_guard_tension_3
		}
		remove_ideas = ROM_iron_guard_tension_3
	}
	else_if = {
		limit = {
			has_idea = ROM_iron_guard_tension_2
		}
		remove_ideas = ROM_iron_guard_tension_2
	}
	else_if = {
		limit = {
			always = yes
		}
		remove_ideas = ROM_iron_guard_tension
	}
}

ROM_decrease_iron_guard_tension = {
	if = {
		limit = {
			has_idea = ROM_iron_guard_tension_2
		}
		swap_ideas = {
			remove_idea = ROM_iron_guard_tension_2
			add_idea = ROM_iron_guard_tension
		}
	}
	else_if = {
		limit = {
			has_idea = ROM_iron_guard_tension_3
		}
		swap_ideas = {
			remove_idea = ROM_iron_guard_tension_3
			add_idea = ROM_iron_guard_tension_2
		}
	}
	else_if = {
		limit = {
			has_idea = ROM_iron_guard_tension_4
		}
		swap_ideas = {
			remove_idea = ROM_iron_guard_tension_4
			add_idea = ROM_iron_guard_tension_3
		}
	}
}

ROM_increase_iron_guard_tension = {
	if = {
		limit = {
			has_idea = ROM_iron_guard_tension_3
		}
		swap_ideas = {
			remove_idea = ROM_iron_guard_tension_3
			add_idea = ROM_iron_guard_tension_4
		}
	}
	else_if = {
		limit = {
			has_idea = ROM_iron_guard_tension_2
		}
		swap_ideas = {
			remove_idea = ROM_iron_guard_tension_2
			add_idea = ROM_iron_guard_tension_3
		}
	}
	else_if = {
		limit = {
			has_idea = ROM_iron_guard_tension
		}
		swap_ideas = {
			remove_idea = ROM_iron_guard_tension
			add_idea = ROM_iron_guard_tension_2
		}
	}
}
